    10      if (remaining.length <= maxCharsPerSegment) {
    11        segments.push(remaining);

    25      segments.push(remaining.substring(0, cutPoint).trim());

    27    }

    33   * 去除 Markdown 格式化符号，返回纯文本。
    34   * 处理：粗体/斜体/标题/列表/引用/代码/删除线。
    35   */
    36  export function stripMarkdown(text: string): string {
    37    if (!text) return text;
    38  
    39    let result = text;
    40  
    41    // 逐行处理行级格式
    42    result = result
    43      .split('\n')
    44      .map(line => {
    45        let l = line.trim();
    46  
    47        // 标题: ### / ## / # 开头
    48        l = l.replace(/^#{1,6}\s+/, '');
    49  
    50        // 无序列表: - / * / + 开头
    51        l = l.replace(/^[-*+]\s+/, '');
    52  
    53        // 有序列表: 1. / 1) / 1、 开头
    54        l = l.replace(/^\d+[.)、]\s*/, '');
    55  
    56        // 引用: > 开头（多层 > >）
    57        l = l.replace(/^(>+\s*)+/, '');
    58  
    59        return l;
    60      })
    61      .join('\n');
    62  
    63    // 内联格式
    64    // 粗体: **text** 或 __text__
    65    result = result.replace(/\*\*(.+?)\*\*/g, '$1');
    66    result = result.replace(/__(.+?)__/g, '$1');
    67  
    68    // 斜体: *text* 或 _text_（但不匹配 ** 残余）
    69    result = result.replace(/(?<!\*)\*(?!\*)(.+?)(?<!\*)\*(?!\*)/g, '$1');
    70    result = result.replace(/(?<!_)_(?!_)(.+?)(?<!_)_(?!_)/g, '$1');
    71  
    72    // 删除线: ~~text~~
    73    result = result.replace(/~~(.+?)~~/g, '$1');
    74  
    75    // 行内代码: `text`
    76    result = result.replace(/`(.+?)`/g, '$1');
    77

    78    // 链接: [text](url) → text
    79    result = result.replace(/\[(.+?)\]\(.+?\)/g, '$1');
    80  
    81    // 图片: ![alt](url) → alt
    82    result = result.replace(/!\[(.+?)\]\(.+?\)/g, '$1');
